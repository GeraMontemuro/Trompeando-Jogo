#include <iostream>
#include <SFML/Graphics.hpp> // Biblioteca de gr치ficos

using namespace std;
using namespace sf;

int main()
{
    RenderWindow window(VideoMode(800, 600), "Beyblade"); // Crea la ventana del juego

    CircleShape beyblade(50.f); // Crea la forma del Beyblade
    beyblade.setFillColor(Color::Red); // Establece el color del Beyblade
    beyblade.setOrigin(50.f, 50.f); // Establece el punto de origen del Beyblade

    Vector2f direction(1.f, 0.f); // Establece la direcci칩n inicial del Beyblade
    float speed = 5.f; // Establece la velocidad de rotaci칩n del Beyblade

    while (window.isOpen()) // Bucle principal del juego
    {
        Event event;
        while (window.pollEvent(event)) // Manejo de eventos
        {
            if (event.type == Event::Closed)
                window.close();
        }

        beyblade.move(direction * speed); // Mueve el Beyblade en la direcci칩n establecida

        window.clear();
        window.draw(beyblade); // Dibuja el Beyblade en la ventana
        window.display();
    }

    return 0;
}
